## P0-2: 진도 추적 (1주, 16h)

### 2-1. useProgress 훅 개발

**파일 생성:** `src/hooks/useProgress.js`

```javascript
export function useProgress() {
  const [progress, setProgress] = useState({});

  // localStorage에서 진도 로드
  useEffect(() => {
    const saved = localStorage.getItem('progress');
    if (saved) setProgress(JSON.parse(saved));
  }, []);

  // 관문 완료 표시
  const completeGate = (topicId, gateId) => {
    const newProgress = {
      ...progress,
      [topicId]: {
        ...(progress[topicId] || {}),
        [gateId]: {
          completed: true,
          timestamp: Date.now()
        }
      }
    };
    setProgress(newProgress);
    localStorage.setItem('progress', JSON.stringify(newProgress));
  };

  // 진도율 계산
  const getCompletionRate = (topicId) => {
    const topic = progress[topicId] || {};
    const completed = Object.values(topic).filter(g => g.completed).length;
    return { completed, total: 7, rate: (completed / 7) * 100 };
  };

  // 전체 진도율
  const getOverallRate = () => {
    let total = 0, completed = 0;
    Object.values(progress).forEach(topic => {
      Object.values(topic).forEach(gate => {
        total++;
        if (gate.completed) completed++;
      });
    });
    return { completed, total: 49, rate: (completed / 49) * 100 };
  };

  return { progress, completeGate, getCompletionRate, getOverallRate };
}
```
